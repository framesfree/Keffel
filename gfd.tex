\begin{document}
    
\begin{tikzpicture}
    \draw[red,thick,dashed] (0,-\Height) -- (0,-\Bullet); %Left wall
    \draw[red,thick,dashed] (\Diam,-\Height) -- (\Diam,-\Bullet); %Right wall

     -- (\Diam-\Bullet,0) -- (\Diam,0-\Bullet); %bulletize right

    \draw[gray,thin] (\Diam/2 - \FCD /2,-\Height) -- (\GrooveExt/2,-\Height) -- (\GrooveExt/2,-\Height+\WO) -- 
      (\GrooveInt/2,-\Height+\WO) -- (\GrooveInt/2,-\Height) -- (0,-\Height); %Groove left
    \draw[blue,thick](\Diam/2 - \FCD /2,-\Height) -- (\Diam/2 + \FCD /2,-\Height); % Flat Contact
